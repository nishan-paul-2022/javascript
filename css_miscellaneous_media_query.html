            @media optional-quantifier optional-media-type and ((media-feature) quantifier (media-feature))

            optional-quantifier
                - not
                - only (making query older browser compatible)
                
            optional-media-type
                - screen
                - speech
                - print
                - all
                
            media-feature
                - min-width: 100px
                - max-width: 100px
                - min-height: 100px
                - max-height: 100px
                - orientation: landascape
                - orientation: portait
                
            quantifier
                - and
                - or
                
            @media screen and (min-width: 600px) and (max-width: 900px), (min-width: 1100px)
            @media screen and ((min-width: 600px) and (max-width: 900px)), (min-width: 1100px)
            @media screen and (((min-width: 600px) and (max-width: 900px)) or (min-width: 1100px))
            
            @media screen and (min-width: 400px) and (max-width: 800px)
            @media screen and (min-width: 400px) and (not (min-width: 800px))
            
            [ optional-quantifier can be used only with optional-media-type ]
